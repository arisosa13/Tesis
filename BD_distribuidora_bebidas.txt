CREATE   PROCEDURE [dbo].[SP_AgregarCliente]
    @NombreCliente VARCHAR(150),
    @Cuit VARCHAR(20),
    @Telefono VARCHAR(30) = NULL,
    @Email VARCHAR(150) = NULL,
    @Direccion VARCHAR(200) = NULL,
    @Ciudad VARCHAR(100) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS (SELECT 1 FROM Clientes WHERE Cuit = @Cuit)
    BEGIN
        RAISERROR('Ya existe un cliente con ese CUIT.', 16, 1);
        RETURN;
    END

    INSERT INTO Clientes (
        NombreCliente,
        Cuit,
        Telefono,
        Email,
        Direccion,
        Ciudad
    )

CREATE   PROCEDURE [dbo].[SP_AgregarDetallePedido]
    @IdPedido INT,
    @IdProducto INT,
    @Cantidad INT,
    @IdUsuario INT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @Precio DECIMAL(10,2);
    DECLARE @StockActual INT;

    IF NOT EXISTS (SELECT 1 FROM Pedidos WHERE IdPedido = @IdPedido)
    BEGIN
        RAISERROR('El pedido no existe.', 16, 1);
        RETURN;
    END

    IF NOT EXISTS (SELECT 1 FROM Productos WHERE IdProducto = @IdProducto)
    BEGIN
        RAISERROR('El producto no existe.', 16, 1);
        RETURN;
    END

    IF NOT EXISTS (SELECT 1 FROM Usuarios WHERE IdUsuario = @IdUsuario)
    BEGIN
        RAISERROR('El usuario no existe.', 16, 1);
        RETURN;
    END

    SELECT
        @Precio = Precio,
        @StockActual = StockActual
    FROM Productos
    WHERE IdProducto = @IdProducto;

    IF @StockActual < @Cantidad
    BEGIN
        RAISERROR('No hay suficiente stock.', 16, 1);
        RETURN;
    END

    IF EXISTS (
        SELECT 1
        FROM DetallePedidos
        WHERE IdPedido = @IdPedido AND IdProducto = @IdProducto
    )
    BEGIN
        UPDATE DetallePedidos
        SET 
            Cantidad = Cantidad + @Cantidad,
            Subtotal = (Cantidad + @Cantidad) * PrecioUnitario
        WHERE IdPedido = @IdPedido
          AND IdProducto = @IdProducto;
    END
    ELSE
    BEGIN
        INSERT INTO DetallePedidos (
            IdPedido,
            IdProducto,
            Cantidad,
            PrecioUnitario,
            Subtotal
        )
        VALUES (
            @IdPedido,
            @IdProducto,
            @Cantidad,
            @Precio,
            @Cantidad * @Precio
        );
    END

    UPDATE Productos
    SET StockActual = StockActual - @Cantidad
    WHERE IdProducto = @IdProducto;

    UPDATE Pedidos
    SET Total = (
        SELECT ISNULL(SUM(Subtotal), 0)
        FROM DetallePedidos
        WHERE IdPedido = @IdPedido
    )
    WHERE IdPedido = @IdPedido;

    INSERT INTO MovimientosStock (
        IdProducto,
        TipoMovimiento,
        Cantidad,
        Motivo,
        IdUsuario
    )
    VALUES (
        @IdProducto,
        'Salida',
        @Cantidad,
        'Salida por pedido N° ' + CAST(@IdPedido AS VARCHAR(20)),
        @IdUsuario
    );

    SELECT 'Detalle procesado correctamente y movimiento registrado' AS Mensaje;
END
GO

CREATE   PROCEDURE [dbo].[SP_ListarProductos]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT 
        p.IdProducto,
        p.NombreProducto,
        p.Marca,
        p.Presentacion,
        p.Precio,
        p.StockActual,
        p.StockMinimo,
        c.NombreCategoria,
        pr.RazonSocial AS Proveedor,
        p.Activo,
        p.FechaAlta
    FROM Productos p
    INNER JOIN Categorias c ON p.IdCategoria = c.IdCategoria
    INNER JOIN Proveedores pr ON p.IdProveedor = pr.IdProveedor
    ORDER BY p.NombreProducto;
END

CREATE   PROCEDURE [dbo].[SP_VerPedidoDetalle]
    @IdPedido INT
AS
BEGIN
    SET NOCOUNT ON;

    SELECT 
        p.IdPedido,
        c.NombreCliente,
        u.Nombre AS Usuario,
        p.FechaPedido,
        p.Estado,
        p.Total
    FROM Pedidos p
    INNER JOIN Clientes c ON p.IdCliente = c.IdCliente
    INNER JOIN Usuarios u ON p.IdUsuario = u.IdUsuario
    WHERE p.IdPedido = @IdPedido;

    SELECT 
        d.IdDetallePedido,
        pr.NombreProducto,
        d.Cantidad,
        d.PrecioUnitario,
        d.Subtotal
    FROM DetallePedidos d
    INNER JOIN Productos pr ON d.IdProducto = pr.IdProducto
    WHERE d.IdPedido = @IdPedido;
END
